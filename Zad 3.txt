function sortowanie_bąbel(arr){
    for(let i = 0; i<arr.length; i++) {
        for(let j = 0; j <(arr.length - i-1);j++) {
            if(arr[j] < arr[j+1]) {
                var temp = arr[j]
                arr[j] = arr[j+1]
                arr[j+1] = temp
        }
    
    }
}
console.log(arr);
}
var arr = [36,28,63,37,25,13,4,6,18];

sortowanie_bąbel(arr);
